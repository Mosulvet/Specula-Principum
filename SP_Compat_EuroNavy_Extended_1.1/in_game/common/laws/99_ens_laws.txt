#European Naval Superiority - Trade Enforcement & Strategic Focus Laws

#The cartaz was a mandatory naval trading license — all vessels needed one or faced seizure
#Mare clausum — closed seas doctrine for trade monopoly
#These laws interact with the land_vs_naval and mercantilism_vs_free_trade societal value axes

ens_trade_enforcement_law = {
	law_category = military
	potential = {
		OR = {
			num_of_ports > 0
			has_law = ens_trade_enforcement_law
		}
	}

	#The cartaz was a mandatory naval trading license — all vessels needed one or faced seizure
	ens_cartaz_system = {
		unique = yes
		potential = {
			OR = {
				has_or_had_tag = POR
				AND = {
					has_or_had_tag = SPA
					culture = culture:portuguese
				}
			}
		}
		country_modifier = {
			trade_range_modifier = 0.15
			blockade_efficiency = 0.15
			merchant_power_from_maritime_modifier = 0.1
			monthly_towards_mercantilism = societal_value_monthly_move
		}
		estate_preferences = {
			burghers_estate
		}
		years = 2
	}

	#Maritime sovereignty — closed seas doctrine for trade monopoly
	ens_mare_clausum = {
		unique = yes
		potential = {
			OR = {
				has_or_had_tag = NED
				has_or_had_tag = ENG
				has_or_had_tag = GBR
			}
		}
		country_modifier = {
			trade_efficiency = small_trade_efficiency_bonus
			global_maritime_presence_modifier = 0.1
			monthly_towards_mercantilism = societal_value_monthly_move
		}
		estate_preferences = {
			burghers_estate
		}
		years = 2
	}

	#Allow free passage of trade vessels
	ens_open_waters_trade = {
		country_modifier = {
			trade_range_modifier = 0.05
			monthly_towards_free_trade = societal_value_minor_monthly_move
		}
		estate_preferences = {
			burghers_estate
		}
		years = 2
	}

	#Dedicated patrols to protect trade routes
	ens_naval_trade_patrols = {
		country_modifier = {
			anti_piracy_warfare_modifier = 0.15
			blockade_efficiency = 0.05
			monthly_towards_outward = societal_value_minor_monthly_move
		}
		estate_preferences = {
			crown_estate
		}
		years = 2
	}
}
