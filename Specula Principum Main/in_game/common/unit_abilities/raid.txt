raid = {
	toggle = no
	army_only = yes

	map = yes

	allow = {
		is_army   = yes
		in_combat = no
		is_moving = no
		is_exiled = no
		in_siege  = no
		num_subunits > 5
		unit_has_leader = yes
		unit_location = {
			development > 5
			owner = {
				is_at_war_with = root.owner
			}
		}
	}


	start_effect = {



		if = {
					limit = {
						current_year > 1690
				}
		raid_effect_doubled = yes

			}
		else = {
			raid_effect = yes
						}




		root.owner = {
			save_scope_as = raid_raider
		}
			damage_unit_percent = 0.005
			damage_unit_morale_percent = 0.1

		unit_location = {

			save_scope_as = raid_location
			add_location_modifier = {
				modifier = raidedmodifier
				years = 1
				mode = add_and_extend
			}


root.owner = {
	add_gold = {
		value    = scope:raid_location.development
		multiply = 1.35
		min      = 2
		max      = 500
	}


	if = {
		limit = {
			scope:raid_location = {
				OR = {
					goods_output = { goods = goods:goods_gold value >= 1 }
					goods_output = { goods = goods:silver    value >= 1 }
				}
			}
		}


		add_gold = {
			value    = scope:raid_location.development
			multiply = 0.4
			min      = 1
			max      = 150
		}
	}
}



			change_prosperity = {
				value    = 0.125
				multiply = -1
			}
			change_development = {
				value    = scope:raid_location.development
				divide = 22
				add = 0.5
				min      = 1
				max      = 5
				multiply = -1
			}

			owner = {
				add_war_exhaustion = {
				value = scope:raid_location.development
				divide = 180
				min = 0.15
				max = 0.6
						}

				add_opinion = {
					modifier = raid_resentment
					target   = scope:raid_raider
					scale    = { value = 1 }
				}
				if = {
					limit = { is_at_war_with = scope:raid_raider }

					"war_with_country(scope:raid_raider)" = {
						add_bonus_warscore = {
							country = scope:raid_raider
							amount  = 1
						}
					}
				}

				if = {
					limit = {
						is_human = yes
						NOT = { has_variable = raid_notifications_disabled }
					}
					trigger_event_non_silently = {
						id = raid_events.1
					}
				}
			}
		}
	}

	ai_will_do = {

		add = -20
}
