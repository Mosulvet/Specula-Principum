kidnap = {
	toggle = no
	army_only = yes
	map = yes

	allow = {
		is_army   = yes
		in_combat = no
		is_moving = no
		is_exiled = no
		in_siege  = no
		num_subunits > 3 #5
		unit_has_leader = yes
		
		dominant_culture = { has_culture_group = culture_group:maghrebi_group } #this was commented out, re-adding.

		unit_location = {
			development > 5
			owner = { is_at_war_with = root.owner }


			NOT = { has_location_modifier = kidnap_recently_kidnapped }
		}
	}

	start_effect = {

		custom_tooltip = {
			text = scorched_earth_pops_unhappy
			every_location_in_province = {
				every_pop = {
					add_pop_satisfaction = pop_satisfaction_severe_penalty
				}
			}
		}

		if = {
			limit = { current_year > 1690 }
			kidnap_effect_doubled = yes
		}
		else = {
			kidnap_effect = yes
		}


		root.owner = { save_scope_as = raid_raider }

		damage_unit_percent = 0.001
		damage_unit_morale_percent = 0.05

		unit_location = {

			save_scope_as = raid_location


			add_location_modifier = {
				modifier = kidnap_recently_kidnapped
				months   = 24
				mode     = add_and_extend
			}


			root.owner = {
				add_prestige       = -0.25
				add_stability      = -0.30
				add_army_tradition = -0.05
			}



			root.owner = {
				capital = {
					save_scope_as = kidnap_destination_location
				}
			}


			custom_tooltip = {
				text = kidnap_destination_tooltip
			}


			add_migration = {
				from_location = scope:raid_location
				to_location   = scope:kidnap_destination_location
				months = 1
				amount = {
					value    = scope:raid_location.population
					multiply = 0.15
					min      = 1
				}
			}


			change_prosperity = {
				value    = 0.10
				multiply = -1
			}

			owner = {
				add_war_exhaustion = {
					value  = scope:raid_location.development
					divide = 180
					min    = 0.05
					max    = 0.3
				}

				add_opinion = {
					modifier = kidnap_resentment
					target   = scope:raid_raider
					scale    = { value = 1 }
				}
				scope:kidnap_destination_location = {
				add_location_modifier = {
					modifier = newarrivals
					months   = 12
					mode     = add_and_extend
				}
			}
				if = {
					limit = { is_at_war_with = scope:raid_raider }

					"war_with_country(scope:raid_raider)" = {
						add_bonus_warscore = {
							country = scope:raid_raider
							amount  = 0.5
						}
					}
				}

				if = {
					limit = {
						is_human = yes
						NOT = { has_variable = raid_notifications_disabled }
					}
					trigger_event_non_silently = { id = raid_events.1 }
				}
			}
		}
	}

	ai_will_do = {
		add = -20
	}
}
