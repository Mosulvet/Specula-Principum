#Chauvache
pillage = {
	toggle = no
	army_only = yes
	map = yes

	allow = {
		is_army   = yes
		in_combat = no
		is_moving = no
		is_exiled = no
		in_siege  = no
		num_subunits > 8
		unit_has_leader = yes

		unit_location = {
			development > 15
			NOT = { has_location_modifier = pillagedmodifier }
			owner = {
				is_at_war_with = root.owner
			}
		}
	}

	start_effect = {

		if = {
			limit = { current_year > 1690 }
			pillage_effect_doubled = yes
		}
		else = {
			pillage_effect = yes
		}


		root.owner = {
			save_scope_as = raid_raider
		}

		damage_unit_percent = 0.0125
		damage_unit_morale_percent = 0.125

		unit_location = {

			save_scope_as = raid_location

			add_location_modifier = {
				modifier = pillagedmodifier
				years    = 2
				mode     = add_and_extend
			}







			root.owner = {
				add_gold = {
					value    = scope:raid_location.development
					multiply = 2.00
					min      = 20
					max      = 800
				}


				if = {
					limit = {
						scope:raid_location = {
							OR = {
								goods_output = { goods = goods:goods_gold value >= 1 }
								goods_output = { goods = goods:silver    value >= 1 }
							}
						}
					}

					add_gold = {
						value    = scope:raid_location.development
						multiply = 0.7
						min      = 15
						max      = 250
					}
				}
			}


			change_prosperity = {
				value    = 0.15
				multiply = -1
			}

			change_development = {
				value    = scope:raid_location.development
				divide   = 20
				add      = 0.6
				min      = 1
				max      = 8
				multiply = -1
			}

			owner = {
				add_war_exhaustion = {
					value  = scope:raid_location.development
					divide = 150
					min    = 0.25
					max    = 1
				}

				add_opinion = {
					modifier = raid_resentment
					target   = scope:raid_raider
					scale    = { value = 1 }
				}

				if = {
					limit = { is_at_war_with = scope:raid_raider }

					"war_with_country(scope:raid_raider)" = {
						add_bonus_warscore = {
							country = scope:raid_raider
							amount  = 3
						}
					}
				}

				if = {
					limit = {
						is_human = yes
						NOT = { has_variable = raid_notifications_disabled }
					}
					trigger_event_non_silently = {
						id = raid_events.1
					}
				}
			}
		}
	}

	ai_will_do = {
		add = -20
	}
}
