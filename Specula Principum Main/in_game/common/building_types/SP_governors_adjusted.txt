TRY_REPLACE:local_governor = {
	is_foreign = no
	pop_type = nobles
	max_levels = 1
	important_for_UI = yes
	
	employment_size = large_noble_employment
	
	category = government_category
	
	rural_settlement = no
	town = yes #no
	city = yes
	megalopolis = yes

	allow = {
		custom_tooltip = {
			text = local_governor_tt
			scope:actor = {
				total_building_levels_including_construction:local_governor < {
					value = modifier:num_local_governors
					subtract = total_building_levels_including_construction:lieutenancy
				}
			}
		}
		owner = scope:actor #not in subjects
		trigger_if = { 
			limit = { has_building = building_type:seljuk_mint }
			NOT = { has_building = building_type:seljuk_mint }
		}
	}

	location_potential = {
		has_road_to_capital = yes
		owner.capital != this
	}

#	remove_if = {
#		location = { has_road_to_capital = no }
#		NOT = { owner = { at_war = yes } }
#		NOT = { owner = { in_civil_war = yes } }
#	}

	expensive = yes #flip to 'no' if AI has problems building
	
	automation_build_allowed = no
	ai_forbid_shutdown = yes
	forbidden_for_estates = yes
	important_for_AI = yes

	build_time = government_build_time


# SP reduced Gold maintenance to encourage AI to build.

	possible_production_methods = {
			local_gov_maintenance_gold
			local_gov_maintenance_silver
		}

	construction_demand = local_governor_construction

# SP 80 was too powerful
	modifier = {
		local_proximity_source = 60 #80
	}
}
