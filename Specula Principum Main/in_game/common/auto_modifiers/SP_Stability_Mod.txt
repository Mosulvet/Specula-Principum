#intent here is to make stability more important, and stability to function like an efficiency drag on council.

TRY_INJECT:positive_stability_impact = {
	
	country_cabinet_efficiency = 0.1 #small bonus	
	global_pop_promotion_speed_modifier = -0.5 #was 1.0
	
	}

TRY_INJECT:negative_stability_impact = {
	potential_trigger = {
		stability < 0
	}	

	scales_with = {
		value = stability
		multiply = -0.01
	}

	global_pop_promotion_speed_modifier = -1.0 #pop promotion slows as stability drops. Note values are multiplied by -0.1.
	country_cabinet_efficiency = -0.5 #as does cabinet efficiency.
	pop_join_rebel_threshold = 0.25	#was 0.1
	trade_efficiency = -0.15 #SP added

}
