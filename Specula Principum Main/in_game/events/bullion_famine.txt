namespace = bullion_famine
bullion_famine.1 = {
	type = country_event
	fire_only_once = yes
	title = bullion_famine.1.title
	desc = bullion_famine.1.desc

	trigger = {
		current_year >= 1360
		capital.continent = continent:europe
		OR = {
			produced_in_country:goods_gold > 0
			produced_in_country:silver > 0
		}
	}

	image = "gfx/interface/illustrations/event/backgrounds/special/shortage.dds"

    option = {
        name = bullion_famine.1.a
		every_other_country = {
			limit = {
				capital = {
					continent = continent:europe
				}
				OR = {
					produced_in_country:goods_gold > 0
					produced_in_country:silver > 0
				}
			}
			trigger_event_silently = bullion_famine.10
		}
		capital.continent = {
			every_location_in_continent = {
				limit = {
					owner ?= root
					OR = {
						raw_material = goods:goods_gold
						raw_material = goods:silver
					}
				}
				add_location_modifier = {
					modifier = the_great_bullion_famine
					years = -1
					mode = add_and_extend
				}
			}

		}
	}
}
