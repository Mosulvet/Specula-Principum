# These changes do several things.
# Reducing cost to build, and restrictsions, to encourage AI to build.
# Temporarily, until proper paradox fix, the land gov building is not deleted when a connection is deleted; or if a naval gov gets land connection. This happens too easily. Naval still disapears when a land connection is created.
# Second, the overall prox bonus is reduced from 80. This could be used with the mod "Local Governors from Land Value", but does not need to be.
# Buildale in towns as well to encourage AI to build.

# TODO - bug multiple maintenance methods showing in game. Try change to REPLACE: or try only changing  key elementvs via INSERT?

TRY_REPLACE:local_governor = {
	is_foreign = no
	pop_type = nobles
	max_levels = 1
	important_for_UI = yes
	
	employment_size = large_noble_employment
	
	category = government_category
	
	rural_settlement = no
	town = yes 
	city = yes
		 
	allow = {
		custom_tooltip = {
			text = local_governor_tt
			scope:actor = {	total_building_levels_including_construction:local_governor < modifier:num_local_governors }
		}
		owner = scope:actor #not in subjects
	}

	location_potential = {
		has_road_to_capital = yes
		owner.capital != this
	}

	expensive = no # cost isn't high regardless, setting to no to help encourage AI to build.
	
	automation_build_allowed = no
	ai_forbid_shutdown = yes
	forbidden_for_estates = yes

	build_time = government_build_time

#removed Gold maintenance to encourage AI to build.

	unique_production_methods = {
		local_governor_maintenance = {
			tools = 1
			masonry = 1
			weaponry = 0.5
			paper = 3
			books = 1
			furniture = 2
			fine_cloth = 1

			category = building_maintenance
		}
	}

	construction_demand = local_governor_construction

#80 was too powerful, especially considering that AI wasn't building
	modifier = {
		local_proximity_source = 60 
	}
}
