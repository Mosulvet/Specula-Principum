country_size_impact_small_human = {
	potential_trigger = {
		is_human = yes
		country_economical_base > 100 # Adds a cut off, below which no impact
		country_economical_base <= 500 #500
		}
		
	scales_with = {
		value = country_economical_base
		subtract = 100
		multiply = 0.015 # was 0.2 Reduces impact at smallest size
	}
	
	court_spending_cost = 0.0075 #added; call this the conspicuous consumption tax
	tax_income_efficiency = -0.015
	#trade_efficiency = -0.01
	#global_production_efficiency = -0.01
	#global_monthly_food_modifier = -0.01
	legislative_efficiency = -0.0125
	country_cabinet_efficiency = -0.0125
	global_max_control = -0.0033 # reduce control impact by 2/3rds
	stability_investment = -0.0125
	#global_estate_target_satisfaction = -0.01
	#global_crown_estate_power = -0.01
	#research_speed_modifier = -0.01
	diplomatic_reputation = -0.1
	antagonism_received_modifier = 0.01
}
