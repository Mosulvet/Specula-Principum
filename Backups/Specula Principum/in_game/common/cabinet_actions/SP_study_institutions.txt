# Should be far more balanced than the flat +1 institution growth to any location in vanilla.
# With this change, institutions should still take a while to spread around the world in the mid-late game. 

study_institutions = {
	ability = adm

	potential = {
		num_locations > 0 
	}

	allow = {
		NOT = { government_type = government_type:tribe }
	}
	
	country_modifier = {
		global_institution_growth_modifier = 0.50
	}
}
